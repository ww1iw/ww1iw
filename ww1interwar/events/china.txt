# Xinhai Revolution breaks out
country_event = {
	id = ww1_china.1
	title = ww1_china.1.t
	desc = ww1_china.1.d
	picture = GFX_report_event_gathering_protest
	
	fire_only_once = yes
	
	trigger = {
		tag = QNG
		has_idea = wanning_empire
		has_idea = zaifeng_regency
		QNG = { controls_state = 622 }
		QNG = { controls_state = 620 }
		QNG = { controls_state = 602 }
		NOT = {
			AND = {
				country_exists = CHI
				has_global_flag = xinhai_revolution
			}
		}
	}
	
	mean_time_to_happen = {
		days = 300
	}

	immediate = {
		hidden_effect = {
			random_list = {
				80 = { 
					620 = { add_core_of = CHI }
				}
				10 = {
					602 = { add_core_of = CHI }
				}
				10 = {
					622 = { add_core_of = CHI }
				}
			}
			release = CHI
			QNG = {
				add_state_core = 620
				add_state_core = 622
				add_state_core = 620
			}
		}
		declare_war_on = { target = CHI type = annex_everything }
		every_owned_state = { 
			add_core_of = CHI
		}
		set_global_flag = xinhai_revolution
		CHI = { load_oob = Xinhai_Hubei }
	}

	option = { 
		name = ww1_china.1.a
	}
}
